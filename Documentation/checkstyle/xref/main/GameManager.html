4   import applogic.UserInputListener;
5   import gui.NextTetriminoGUI;

8   import javax.swing.*;
9   import java.awt.*;
10  
11  /**
12   * Class for managing the game of Tetris.
13   */
14  public class GameManager {
15  
16      private final PlayingFieldGUI playingFieldGUI;
17      private JFrame app;
18      private JLabel scoreLabel;
19      private int delay;
20  
21      /**
22       * Sets up a standard sized playing field and new JFrame
23       * and parameters of to it. Then adds the playing field and next
24       * Tetrimino shower to the JFrame and starts the game.
25       * @throws InterruptedException is thrown if Thread.sleep() method is interrupted.
26       */
27      public GameManager() throws InterruptedException {
28          playingFieldGUI = new PlayingFieldGUI(20, 10);
29          app = new JFrame();
30          newApp();
31          delay = 400;
32          start();
33      }
34  
35      /**
36       * Method that sets up the JFrame parameters.
37       */
38      private void newApp() {
39          app.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
40          app.setLayout(new FlowLayout());
41          app.setSize(600, 600);
42          app.setTitle("Yet Another Tetris Clone");
43          app.addKeyListener(new UserInputListener(this.playingFieldGUI));
44          app.add(playingFieldGUI);
45          app.add(new NextTetriminoGUI(playingFieldGUI));
46          scoreLabel = new JLabel(getScore());
47          app.add(scoreLabel);
48      }
49  
50      /**
51       * Method that starts the game and keeps it going until game over.
52       * Additionally it checks if the difficulty level should be increased.
53       * @throws InterruptedException Thrown if Thread.sleep() is interrupted.
54       */
55      public void start() throws InterruptedException {
56          app.setVisible(true);
57          playingFieldGUI.addNewTetrimino();
58          playingFieldGUI.repaintPlayingField();
59  
60          while (!playingFieldGUI.gameOver()) {
61              scoreLabel.setText(getScore());
62              if (playingFieldGUI.getPlayingField().getLevelCounter() >= 10) {
63                  shortenTheDelay(); // Increases the difficulty by shortening the time that the thread sleeps.
64                  playingFieldGUI.getPlayingField().resetNumberOfRemovedRows();
65              }
66              playingFieldGUI.tick();
67              Thread.sleep(delay);
68          }
69          app.removeAll();
70          app.dispose();
71  
72      }
73  
74      /**
75       * Gets the playing field.
76       * @return PlayingFieldGUI object.
77       */
78      public PlayingFieldGUI getPlayingField() {
79          return playingFieldGUI;
80      }
81  
82      /**
83       * Method that sortens the delay by 25 milliseconds. Used when level
84       * of difficulty is increased.
85       */
86      public void shortenTheDelay() {
87          this.delay = this.delay - 25;
88      }
89  
90      /**
91       * Method for getting the score that the player has.
92       * @return Score óf the player as int.
93       */
94      public String getScore() {
95          return this.playingFieldGUI.getPlayingField().getScore();
96      }
97  }
