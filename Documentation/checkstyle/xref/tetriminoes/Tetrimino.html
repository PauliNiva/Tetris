5   /**
6    * Abstract class for all Tetrimino pieces.
7    */
8   public abstract class Tetrimino {
9   
10      /**
11       * Tableaux for holding a Tetrimino piece in, ergo Tetrimino piece holder.
12       */
13      final int[][] cells;
14  
15      /**
16       * Creates a new Tetrimino blocks by first making a 4 by 4 empty tableaux and then sets
17       * the corresponding cells that are occupied by the specific Tetrimino piece to a value 1.
18       * Value 1 marks that the cell contains one block of the specified Tetrimino and 0 marks that
19       * it does not.
20       */
21      Tetrimino() {
22          cells = new int[4][4];
23          setTetriminoInCells();
24      }
25  
26      /**
27       * Abstract method without an implementation that sets a specific Tetrimino inside the cell tableaux.
28       */
29      abstract void setTetriminoInCells();
30  
31      /**
32       * Method that gets the layout of the Tetrimino piece inside the cell.
33       * This method is used to get the orientation of the piece inside the cells.
34       * @return cell tableaux that contains one Tetrimino piece.
35       */
36      public int[][] getLayout() {
37          return cells;
38      }
39  
40      /**
41       * Abstract method without an implementation that is used to get the color of a specific Tetrimino piece.
42       * Tetrimino piece colors adhere to the Tetris companys standardization.
43       * @return Color of the Tetrimino piece.
44       */
45      public abstract Color getColor();
46  }
