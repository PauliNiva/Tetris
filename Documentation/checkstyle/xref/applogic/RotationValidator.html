4   import tetriminoes.Tetrimino;
5   
6   /**
7    * Class for validating the rotations of the Tetrimino piece.
8    */
9   public class RotationValidator {
10  
11      PlayingField playingField;
12      Mapper mapper;
13      Tetrimino tetrimino;
14  
15      /**
16       * Creates a RotationValidator object with the currently falling Tetrimino,
17       * Playingfield and it's mapper at their current state.
18       * @param playingField PlayingField object at it's current state.
19       * @param tetrimino Tetrimino piece that is currently falling in the field.
20       * @param mapper Mapper at it's current state.
21       */
22      public RotationValidator(PlayingField playingField, Tetrimino tetrimino, Mapper mapper) {
23          this.playingField = playingField;
24          this.tetrimino = tetrimino;
25          this.mapper = mapper;
26      }
27  
28      /**
29       * Method that checks if the rotated Tetrimino will end up occupying an already occupied cell.
30       * @param newMatrix Matrix that contains the layout of the rotated Tetrimino container.
31       * @return boolean value true, it the cell is already occupied, false otherwise.
32       */
33      public boolean rotatedTetriminoWillOccupyAnOccupiedCell(int[][] newMatrix) {
34          for (int row = 0; row < newMatrix.length; row++) {
35              for (int column = 0; column < newMatrix[0].length; column++) {
36                  if (mapper.getPlayingfieldCellAt(row, column).getIsCellOccupied() && newMatrix[row][column] == 1 &&
37                          !(mapper.getTetriminoContainerAsList().contains(mapper.getPlayingfieldCellAt(row, column)))) {
38                      return true;
39                  }
40              }
41          }
42          return false;
43      }
44  
45      /**
46       * Method that checks if the rotated Tetrimino would rotate off the playingfield.
47       * @param newMatrix Matrix that contains the layout of the rotated Tetrimino container.
48       * @return boolean value true if the piece would rotate of the field, false otherwise.
49       */
50      public boolean tetriminoWillRotateOffPlayingfield(int[][] newMatrix) {
51          return tetriminoWillRotateOffLeftEdge(newMatrix) ||
52                  tetriminoWillRotateOffRightEdge(newMatrix);
53      }
54  
55      /**
56       * Utility method for checking if the Tetrimino would rotate off the right edge of the playing field.
57       * @param newMatrix Matrix that contains the layout of the rotated Tetrimino container.
58       * @return boolean value true is the rotating Tetrimino would move off the field from right edge,
59       * false otherwise.
60       */
61      public boolean tetriminoWillRotateOffRightEdge(int[][] newMatrix) {
62          return (tetriminoWillMoveToRight(newMatrix) && tetriminoIsInColumn(playingField.getColumns() - 1));
63      }
64  
65      /**
66       * Utility method for checking if the Tetrimino would rotate off the left edge of the playing field.
67       * @param newMatrix Matrix that contains the layout of the rotated Tetrimino container.
68       * @return boolean value true is the rotating Tetrimino would move off the field from left edge,
69       * false otherwise.
70       */
71      private boolean tetriminoWillRotateOffLeftEdge(int[][] newMatrix) {
72          return (tetriminoWillMoveToLeft(newMatrix) && tetriminoIsInColumn(0));
73      }
74  
75      /**
76       * Utility method to check if rotation would move the Tetrimino to the right.
77       * @param newMatrix int[][] tableaux of a layout of an already rotated Tetrimino.
78       * @return boolean value true, if the Tetrimino would move to the right, false otherwise.
79       */
80      private boolean tetriminoWillMoveToRight(int[][] newMatrix) {
81          return matrixContainsOccupiedCellInColumn(newMatrix, 3) &&
82                  !matrixContainsOccupiedCellInColumn(tetrimino.getLayout(), 3);
83      }
84  
85      /**
86       * Utility method to check if rotation would move the Tetrimino to the left.
87       * @param newMatrix int[][] tableaux of a layout of an already rotated Tetrimino.
88       * @return boolean value true, if the Tetrimino would move to the left, false otherwise.
89       */
90      private boolean tetriminoWillMoveToLeft(int[][] newMatrix) {
91          return matrixContainsOccupiedCellInColumn(newMatrix, 0) &&
92                  !matrixContainsOccupiedCellInColumn(tetrimino.getLayout(), 0);
93      }
94  
95      /**
96       * Checks if matrix contains a occupied cell in specified column.
97       * @param newMatrix int[][] tableaux that contains the cells to be checked.
98       * @param column column that is to be checked if it has any occupied matrix cells in it.
99       * @return boolean value true, if matrix has a cell in specified column, false otherwise.
100      */
101     private boolean matrixContainsOccupiedCellInColumn(int[][] newMatrix, int column) {
102         for (int[] row : newMatrix) {
103             if (row[column] == 1) {
104                 return true;
105             }
106         }
107         return false;
108     }
109 
110     /**
111      * Method that checks if Tetrimino is in certain column.
112      * @param column that is to be checked if it does contain piece of Tetrimino piece.
113      * @return boolean value true if column has a piece of Tetrimino in it, otherwise false.
114      */
115     private boolean tetriminoIsInColumn(int column) {
116         for (Cell cell : mapper.getTetriminoContainerAsList()) {
117             if (cell.getColumn() == column) {
118                 return true;
119             }
120         }
121         return false;
122     }

124 }
