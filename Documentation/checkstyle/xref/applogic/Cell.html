1   package applogic;
2   
3   import tetriminoes.Tetrimino;
4   
5   import java.awt.*;
6   
7   /**
8    * Class for the cells which together form the grid called the playing field.
9    */
10  public class Cell {
11  
12      private final int row;
13      private final int column;
14      private Color color;
15  
16      private boolean cellIsOccupied;
17  
18      /**
19       * Makes a new cell in a specific location as dictated by the coordinates from
20       * the parametres.
21       * @param row int that serves as an Y coordinate. That is at which row the cells place is
22       *            in the cell tableaux called the playing field.
23       * @param column int that serves as an X coordinate. That is at which column the cells place is
24       *            in the cell tableaux called the playing field
25       */
26      public Cell(int row, int column) {
27          this.row = row;
28          this.column = column;
29          color = PlayingField.getDEFAULT_EMPTY_COLOR();
30      }
31  
32      /**
33       * Sets that the cell is occupied by Tetrimino block and changes the color of the cell
34       * accordingly.
35       * @param tetrimino Which kind of Tetrimino is occupying the cell.
36       * @param cellIsOccupied boolean value telling that is there a Tetromino block in the cell.
37       * @return information about the color and the occupation of the cell.
38       */
39      public Cell setCellIsOccupied(Tetrimino tetrimino, boolean cellIsOccupied) {
40          this.cellIsOccupied = cellIsOccupied;
41          color = ((cellIsOccupied && tetrimino != null) ? tetrimino.getColor() : PlayingField.getDEFAULT_EMPTY_COLOR());
42  
43          return this;
44      }
45  
46      /**
47       * Method to set cell as occupied with one parameter.
48       * @param tetrimino piece that is occupying the cell.
49       * @return calls setCellIsOccupied method with two parameters.
50       */
51      public Cell setCellIsOccupied(Tetrimino tetrimino) {
52          return setCellIsOccupied(tetrimino, true);
53      }
54  
55      /**
56       * Changes the state of cell occupation (used after the piece
57       * leaves the cell).
58       * @param occupied boolean value that the cells value is to be changed.
59       * @return calls setCellIsOccupied method with two parameters.
60       */
61      public Cell setCellIsOccupied(boolean occupied) {
62          return setCellIsOccupied(null, occupied);
63      }
64  
65      /**
66       * Method for checking if the cell is occupied.
67       * @return boolean value telling if the cell is occupied.
68       */
69      public boolean getIsCellOccupied() {
70          return cellIsOccupied;
71      }
72  
73      /**
74       * Method for getting the color of the cell.
75       * @return color of the cell.
76       */
77      public Color getColor() {
78          return this.color;
79      }
80  
81      /**
82       * Method for setting the color of the cell.
83       * @param color which the cell is to be set.
84       */
85      public void setColor(Color color) {
86          this.color = color;
87      }
88  
89      /**
90       * Gets the number of row.
91       * @return number of row as an int.
92       */
93      public int getRow() {
94          return this.row;
95      }
96  
97      /**
98       * Gets the number of column.
99       * @return number of column as an int.
100      */
101     public int getColumn() {
102         return this.column;
103     }
104 
105     /**
106      * Method that checks if the occupied cell belongs to the same Tetrimino piece
107      * that is moving or another.
108      * @param object Tetrimino as an object.
109      * @return boolean value true if the cell belongs to the same Tetrimino, false otherwise.
110      */
111     @Override
112     public boolean equals(Object object) {
113         if (this == object) {
114             return true;
115         }
116         if (object == null || getClass() != object.getClass()) {
117             return false;
118         }
119 
120         Cell cell = (Cell) object;
121 
122         if (column != cell.column) {
123             return false;
124         }
125         if (row != cell.row) {
126             return false;
127         }
128 
129         return true;
130     }
131 }
